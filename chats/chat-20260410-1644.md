 * @param {string} responseText The full response text string, including [Model]: headers and --- separators.

function compressResponseText(responseText, maxCharsPerModel) {
    // Split the response text into individual model responses
    const modelResponses = responseText.split('---');

    // Initialize an empty array to store the compressed model responses
    const compressedModelResponses = [];

    // Iterate over each model response
    for (const modelResponse of modelResponses) {
        // Check if the model response is not empty
        if (modelResponse.trim() !== '') {
            // Extract the model header
            const modelHeader = modelResponse.substring(0, modelResponse.indexOf('\n'));

            // Extract the response text without the model header
            const response = modelResponse.substring(modelResponse.indexOf('\n'));

/**
 * Truncates model responses to a maximum number of characters per model while preserving structure.
 * @param {string} responseText - Full response text with [Model]: headers, and --- separators.

 * @returns {string} - Compressed response text with truncated individual model responses.
 */
function compressResponseText(responseText, maxCharsPerModel) {
  // Split the responses by separators into blocks
  const blocks = responseText.split('---');

  const compressedBlocks = blocks.map((block) => {
    const matchHeader = block.match(/^\[Model\]:\s*([\s\S]*?)(?:\[CODER:/)?/);
    if (!matchHeader) return block.trim(); // If no header, return the block as is.

    const header = matchHeader[0]; // Includes the `[Model]:` part.
    const contentStartIndex = block.indexOf(header) + header.length; // Start of the content.
    const content = block.slice(contentStartIndex).trim();

    const coderTagMatch = content.match(/\[CODER:\s*[^]*$/); // Find [CODER: ...] tag if present.
    const coderTag = coderTagMatch ? coderTagMatch[0] : '';
    const mainContent = coderTagMatch ? content.replace(coderTag, '').trim() : content;

    // Truncate the content to the max allowed characters, while keeping the coderTag if present
    const truncatedContent = mainContent.slice(0, maxCharsPerModel).trim();

    return `${header}${truncatedContent}${coderTag}`;
  });

  // Reassemble the compressed blocks with separators
  return compressedBlocks.join('\n---\n').trim();
}
