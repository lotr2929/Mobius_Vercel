import * as THREE from './three.module.js';

/**
 * Creates a 3D compass needle pointing to true north
 * @param {THREE.Scene} scene - The Three.js scene to add the compass to
 * @param {number} designNorthAngle - Current design north angle in radians
 * @returns {THREE.Group} The compass group containing all visual elements
 */
export function createNorthPoint3D(scene, designNorthAngle) {
    const compassGroup = new THREE.Group();
    compassGroup.name = 'northPoint3D';

    // Compass needle (red arrow)
    const needleGeometry = new THREE.ConeGeometry(0.5, 2, 8);
    const needleMaterial = new THREE.MeshBasicMaterial({ color: 0xff0000 });
    const needle = new THREE.Mesh(needleGeometry, needleMaterial);
    needle.position.y = 1;
    needle.rotation.x = Math.PI / 2; // Point upwards
    compassGroup.add(needle);

    // Compass base (circle)
    const baseGeometry = new THREE.CircleGeometry(1, 32);
    const baseMaterial = new THREE.MeshBasicMaterial({
        color: 0x333333,
        side: THREE.DoubleSide
    });
    const base = new THREE.Mesh(baseGeometry, baseMaterial);
    compassGroup.add(base);

    // Compass label
    const labelCanvas = document.createElement('canvas');
    labelCanvas.width = 128;
    labelCanvas.height = 64;
    const context = labelCanvas.getContext('2d');
    context.fillStyle = 'white';
    context.font = 'Bold 24px Century Gothic';
    context.textAlign = 'center';
    context.fillText('N', 64, 32);

    const labelTexture = new THREE.CanvasTexture(labelCanvas);
    const labelMaterial = new THREE.SpriteMaterial({ map: labelCanvas });
    const label = new THREE.Sprite(labelMaterial);
    label.position.y = 1.5;
    label.scale.set(0.5, 0.25, 1);
    compassGroup.add(label);

    // Position compass in the top-right corner
    compassGroup.position.set(8, 4, 0);

    // Rotate compass to match design north angle
    compassGroup.rotation.y = designNorthAngle;

    scene.add(compassGroup);
    return compassGroup;
}

/**
 * Updates the 3D compass orientation
 * @param {THREE.Group} compassGroup - The compass group to update
 * @param {number} designNorthAngle - New design north angle in radians
 */
export function updateNorthPoint3D(compassGroup, designNorthAngle) {
    if (!compassGroup) return;

    // Smooth rotation animation
    const targetRotation = designNorthAngle;
    const currentRotation = compassGroup.rotation.y;

    // Simple linear interpolation for smooth rotation
    compassGroup.rotation.y = THREE.MathUtils.lerp(
        currentRotation,
        targetRotation,
        0.1
    );
}
