const MAX_CONTEXT_CHARS = 1500;
const CHUNK_LIMIT = 5;

export async function getCodeContext(query) {
  // 1️⃣ sanitise query → glob pattern
  const pattern = query.toLowerCase().replace(/\./g, '\\.').replace(/\*/g, '%').replace(/\?/g, '_');
  const likeClause = `${pattern}%`; // starts‑with match

  // 2️⃣ fetch candidate chunk ids by filename
  const { data: nameRows } = await db
    .from('code_chunks')
    .select('chunk_id')
    .ilike('filename', likeClause);

  if (!nameRows.length) return { context: '', truncated: false };

  const ids = nameRows.map(r => r.chunk_id);

  // 3️⃣ embed the query once
  const { embedding } = await db.rpc('embed_query', [query]);

  // 4️⃣ similarity search limited to candidate ids
  const { data: similarRows } = await db.rpc('match_code_chunks', [
    embedding,
    ids,               // pass array to restrict search
    0.85,              // similarity threshold
    CHUNK_LIMIT        // max rows to return
  ]);

  // 5️⃣ deduplicate & truncate
  const seen = new Set();
  const chunks = [];
  let used = 0;

  for (const row of similarRows) {
    if (seen.has(row.chunk_id)) continue;
    seen.add(row.chunk_id);

    if (toAdd <= 0) break;

    chunks.push(row.code_text.slice(0, toAdd));
    used += toAdd;
    if (used >= MAX_CONTEXT_CHARS) break;
  }

  return {
    context: chunks.join('\n'),
    truncated: used < (similarRows.length * 300), // simple heuristic
    sourceFiles: [...seen].map(id => /* fetch filename */)
  };
}
